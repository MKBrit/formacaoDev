#include <iostream>
#include <locale>
using namespace std;

#define N 5

int main() {
    setlocale(LC_ALL, "portuguese");

    bool a;
    while (a = true)
    {    
        int vetorX[N];
        int vetorY[N];

        cout << "Digite os " << N << " elementos do vetor X:" << endl;
        for (int i = 0; i < N; i++) {
            cin >> vetorX[i];
        }

        cout << "Digite os " << N << " elementos do vetor Y:" << endl;
        for (int i = 0; i < N; i++) {
            cin >> vetorY[i];
        }

        cout << "Posições dos elementos do vetor X maiores que os do vetor Y:" << endl;
        for (int i = 0; i < N; i++) {
            if (vetorX[i] > vetorY[i]) {
                cout << "Posição " << i+1 << ": " << vetorX[i] << endl;
            }
        }

        cout << endl << "0 stop/ 1 go" << endl;
        cin >> a;
        cout << endl;
    }
    return 0;
}
